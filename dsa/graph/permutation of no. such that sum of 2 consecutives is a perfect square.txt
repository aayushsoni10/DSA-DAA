#include<bits/stdc++.h>
using namespace std;
bool solveU(vector<int> &vis, int n, int i, int c){
    if(c==n){
        vis[i] = 0;
        cout<<i<<' ';
        return true;
    }
    for(int j=1;j<=n;j++){
        double f = sqrt(i+j);
        if(f-(int)f==0.0){
            if(!vis[j]){
                vis[j] = 1;
                if(solveU(vis, n, j, c+1)){
                    cout<<i<<' ';
                    return true;
                }
            }
        }
    }
    vis[i] = 0;
    return false;
}
void solve(int n){
    vector<int> vis(n+1, 0);
    for(int i=1;i<=n;i++){
        vis[i] = 1;
        if(solveU(vis, n, i, 1)){
            return;
        }
    }
}
int main(){
    int n;
    cin>>n;
    solve(n);
}
