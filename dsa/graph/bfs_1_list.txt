#include<bits/stdc++.h>
using namespace std;
vector<int> *a;
void bfs(int n, int x){
    int lev[n+1]={0};
    int vis[n+1]={0};
    queue<int> q;
    vis[1]=1;
    lev[1]=1;
    q.push(1);
    while(!q.empty()){
        int v=q.front();
        q.pop();
        for(int i=0;i<(int)a[v].size();i++){
            if(!vis[a[v][i]]){
                vis[a[v][i]]=1;
                lev[a[v][i]]=lev[v]+1;
                q.push(a[v][i]);
            }
        }
    }
    int c=0;
    for(int i=1;i<n+1;i++){
        if(lev[i]==x)
            c++;
    }
    printf("%d",c);
}
int main(){
    int n;
    scanf("%d",&n);
    a=new vector<int>[n+1];
    for(int i=0;i<n-1;i++){
        int u,v;
        scanf("%d %d",&u,&v);
        a[u].push_back(v);
        a[v].push_back(u);
    }
    int x;
    scanf("%d",&x);
    bfs(n,x);
}
