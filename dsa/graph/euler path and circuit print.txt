#include <bits/stdc++.h>

using namespace std;

void U(stack<int> &st, vector<vector<int>> &adj, vector<int> &path, int s){
    st.push(s);
    for(int k=0;k<adj.size();k++){
        if(!adj[s][k])
            continue;
        adj[s][k] = 0;
        adj[k][s] = 0;
        U(st, adj, path, k);
    }
    path.push_back(st.top());
    cout<<st.top()<<' '<<s<<endl;
    st.pop();
}
int main() {
	int n, m;
	cin>>n>>m;
	vector<vector<int>> adj(n, vector<int> (n, 0));
	int s = 0;
	vector<int> vd(n, 0);
	for(int i=0;i<m;i++){
	    int a,b;
	    cin>>a>>b;
	    vd[a]++;
	    vd[b]++;
	    adj[a][b] = 1;    
        adj[b][a] = 1;
	}
	for(int i=0;i<n;i++)
	    if(vd[i]%2){
	        s = i;
	        break;
	    }
	stack<int> st;
	vector<int> path;
	U(st, adj, path, s);
	for(int i=0;i<path.size();i++)
	    cout<<(path[i]+1);
	return 0;
}
