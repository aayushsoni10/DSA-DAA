#include<bits/stdc++.h>

using namespace std;

int main(){
    int n;
    cin>>n;
    queue<string> q;
    q.push("1");
    while(!q.empty()){
        string a = q.front();
        q.pop();
        int val = stoi(a);
        if((val%n) == 0){
            cout<<a<<endl;
            break;
        }
        q.push(a + "0");
        q.push(a + "1");
    }
    return 0;
}
