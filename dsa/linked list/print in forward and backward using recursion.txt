#include <iostream>

using namespace std;
struct Node{
    int data;
    Node* next;
};
void PrintB(Node *head){
    if(head==NULL){
        printf("");
    }else{
        PrintB(head->next);
        printf("%d ",head->data);
    }
}
int main() {
	Node* h1 = new Node();
    h1->data=2;
    h1->next=NULL;
    Node* h = new Node();
    h->data=1;
    h->next=h1;
    PrintB(h);
}
