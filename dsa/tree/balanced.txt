/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
int solU(TreeNode* A){
    if(!A)  return 0;
    int l = solU(A->left);
    int r = solU(A->right);
    if(l==-1 || r==-1)
        return -1;
    if(abs(l-r)>1){
        return -1;
    }
    return max(l,r)+1;
}
int Solution::isBalanced(TreeNode* A) {
    int a = solU(A);
    return a==-1?0:1;
}
