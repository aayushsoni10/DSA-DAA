#include <bits/stdc++.h>
using namespace std;
void knapsack(int n,int val[],int wt[],int a){
    int dp[n][a+1];
    for(int i=0;i<n;i++)
        dp[i][0]=0;
    for(int j=1;j<a+1;j++){
        if(wt[0]<=j){
            dp[0][j]=val[0];
        }else{
            dp[0][j]=0;
        }
    }
    for(int i=1;i<n;i++){
        for(int j=1;j<a+1;j++){
            if(wt[i]>j){
                dp[i][j]=dp[i-1][j];
            }else{
                dp[i][j]=max(val[i]+dp[i-1][j-wt[i]],dp[i-1][j]);
            }
        }
    }
    printf("%d\n",dp[n-1][a]);
    int i=n-1;
    int j=a;
    while(j!=0 && i>=0){
        if(dp[i][j]==dp[i-1][j]){
            i=i-1;
            continue;
        }
        else{
            printf("%d ",val[i]);
            j=j-wt[i];
            i=i-1;
        }
    }
}
int main(){
    int n;
    scanf("%d",&n);
    int val[n];
    int wt[n];
    for(int i=0;i<n;i++){
        scanf("%d",&val[i]);
    }
    for(int i=0;i<n;i++){
        scanf("%d",&wt[i]);
    }
    int a;
    scanf("%d",&a);
    knapsack(n,val,wt,a);
}
