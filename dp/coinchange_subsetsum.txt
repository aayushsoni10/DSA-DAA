// Complete the getWays function below.
long getWays(long n, vector<long> c) {
    long dp[c.size()][n+1];
    int m=c.size();
    for(int i=0;i<m;i++){
        dp[i][0]=1;
    }
    for(int i=1;i<=n;i++){
        if(i<c[0])
            dp[0][i]=0;
        else
            dp[0][i]=dp[0][i-c[0]];
    }
    for(int i=1;i<m;i++){
        for(int j=1;j<=n;j++){
            if(j<c[i])
                dp[i][j]=dp[i-1][j];
            else
                dp[i][j]=dp[i-1][j]+dp[i][j-c[i]];
        }
    }
    return dp[m-1][n];
}

int main()
{
    long n,m;
    scanf("%ld %ld",&n,&m);
    vector<long> c(m);
    for(int i=0;i<m;i++){
        scanf("%ld",&c[i]);
    }
    printf("%ld",getWays(n,c));
    return 0;
}
